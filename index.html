const express = require("express"); const http = require("http"); const WebSocket = require("ws"); const app = express(); const server = http.createServer(app); const wss = new WebSocket.Server({ server }); // DATA MATCH let match = { home: "Canada", away: "Bosnia & Herzegovina", scoreHome: 0, scoreAway: 1, minute: 69, second: 40, status: "LIVE", odds: { handicap_home: 2.10, handicap_away: 1.78 } }; // kirim ke semua client function broadcast() { const data = JSON.stringify(match); wss.clients.forEach(client => { if (client.readyState === 1) { client.send(data); } }); } // SIMULASI LIVE UPDATE setInterval(() => { // update timer match.second++; if (match.second >= 60) { match.minute++; match.second = 0; } // simulasi gol if (Math.random() < 0.03) { Math.random() > 0.5 ? match.scoreHome++ : match.scoreAway++; } // simulasi odds naik turun match.odds.handicap_home = (parseFloat(match.odds.handicap_home) + (Math.random() - 0.5) * 0.05).toFixed(2); match.odds.handicap_away = (parseFloat(match.odds.handicap_away) + (Math.random() - 0.5) * 0.05).toFixed(2); broadcast(); }, 2000); // websocket connection wss.on("connection", (ws) => { ws.send(JSON.stringify(match)); }); // start server server.listen(3000, () => { console.log("LIVE SERVER RUNNING: http://localhost:3000"); });
