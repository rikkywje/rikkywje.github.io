const express = require("express"); const http = require("http"); const WebSocket = require("ws"); const app = express(); const server = http.createServer(app); const wss = new WebSocket.Server({ server }); // DATA MATCH (SEPERTI SOFASCORE) let match = { home: "Canada", away: "Bosnia & Herzegovina", scoreHome: 0, scoreAway: 1, minute: 69, second: 40, status: "LIVE", odds: { handicap_home: 2.10, handicap_away: 1.78 } }; // broadcast ke semua client function sendUpdate() { wss.clients.forEach(client => { if (client.readyState === 1) { client.send(JSON.stringify(match)); } }); } // SIMULASI LIVE UPDATE (score + timer + odds berubah) setInterval(() => { // update waktu match.second++; if (match.second >= 60) { match.minute++; match.second = 0; } // simulasi gol kecil (random) if (Math.random() < 0.03) { if (Math.random() > 0.5) match.scoreHome++; else match.scoreAway++; } // simulasi odds berubah match.odds.handicap_home += (Math.random() - 0.5) * 0.05; match.odds.handicap_away += (Math.random() - 0.5) * 0.05; // rounding match.odds.handicap_home = match.odds.handicap_home.toFixed(2); match.odds.handicap_away = match.odds.handicap_away.toFixed(2); sendUpdate(); }, 2000); // client connect wss.on("connection", ws => { ws.send(JSON.stringify(match)); }); server.listen(3000, () => { console.log("Server running http://localhost:3000"); });
